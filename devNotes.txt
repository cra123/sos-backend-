EmergencyEvent.create({
                typeOfEmergency,
                location,
                geolocation_lat,
                geolocation_lng,
                description,
                imageUrl,
                status,
                user,
            });


        //Return the creted event
        .then((event) => {
            res.status(200).json(event);
        })
        //Push the event to the user
        .then((user, event) => {
            return User.findByIdAndUpdate(user, { $push: { emergencyEvent: event._id } });
        })
        //Return error if any
        .catch((err) => {
            res.status(500).json(err);
        });
